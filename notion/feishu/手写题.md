    for(let i = 0;i<parts.length-1;i+=2){
        const keyPart = parts[i].trim()
        const key = keyPart.slice(0,keyPart.indexOf('=')).trim()

    }

Function.prototype._call = function（context,...args）{
    // 不是function 类型错误
    if(typeof context !== 'function') {
        throw new TypeError('_call must be called on function')
    }
    
    // 确定this对象
    context = (typeof context === 'object' && context !== null ) ? context : window
    
    // 建立唯一标识，避免属性名冲突
    const symbol =new Symbol()
    context[symbol] = this
    
    // 执行函数
    const result = context[symbol](...args)
    
    // 删除唯一标识
    delete context[symbol]
    return result
}

const deppClone(obj，map = new Map()){
    // 基本数据类型或者null 直接返回
    if(typeof obj !== 'object' || obj === null) return obj
    
    // 处理Date、RegExp特殊对象
    let constructor = obj.constructor
    if(/^(Date|RegExp)$/i.test(constructor.name)) return new constructor(obj)
    // funtion 直接返回
    if(obj instanceof funtion) return obj
    
    // 处理map和set
    if(obj instanceof Map) return new Map(
        Array.from(obj.entries()).map(([k,v])=>{deepClone(k,map),deepClone(v,map)}
    )
    if(obj instanceof Set) return new Set(
        Array.from(obj).map(item => deepClone(item,map))
    )
    // 如果再次遇到同一对象，直接返回之前克隆的结果，避免无限递归
    if(map.has(obj)) return map.get(obj)
    
    // 创建新对象或数组
    const cloneObj = Array.isArray(obj) ?[]:{}
    map.set(obj,cloneObj)
    
    //遍历属性递归深拷贝
    for(let key in obj){
        if(obj.hasOwnProperty(key)){
            cloneObj[key]=deepClone(obj[key],map)
        }
    }
    return cloneObj    
}

1.  实现一个快速排序
function quickSort(arr){
    if (arr.length <= 1) return arr;  //递归终止条件 上次错误点
    let n = arr.length
    let preIndex = Math.floor(n/2)
    let preCount = arr[preIndex]
    const left = []
    const right = []
    for (let i = 0; i < arr.length; i++) {
        if (i === preIndex) continue; // 只跳过基准值所在位置的元素
        let item = arr[i];
        if (item < preCount) {
            left.push(item);
        } else {
            right.push(item);
        }
    }    
    
    return [...quickSort(left),preCount,...quickSort(right)]
}

function MyPromise(exector){
    this.status = "pending"
    this.value = undefined
    this.reason = undefined
    this.callBacks = []
    
    const resolve = (value) => {
        if(this.status === "pending"){
            this.status = "fulfilled"
            this.value = value
            this.callBacks.forEach(callBack => callBack.onFulfilled(value))
        }
    }
    
    const reject = (reason) =>{
        if(this.status === "pending"){
            this.status = "rejected"
            this.reason = reason
            this.callBacks.forEach(callBack => callBack.onRejected(reason))
        }
    }
    
    try{
        exector(resolve,reject)
    } catch(error){
        reject(error)
    }
}

        onFulfilled(this.value)
    } else if(this.status === "rejected"){
        onRejected(this.reason)
    } else if(this.status === "pending"){
        this.callBacks.push({onFulfilled,onRejected})
    }
}

64. 使用promise实现限流
async function promisePool(tasks, limit) {
  const pool = new Set();

  for (const task of tasks) {
    const p = task().then(() => pool.delete(p));
    pool.add(p);

    if (pool.size >= limit) {
      await Promise.race(pool); // 等到任意一个任务结束再加新任务
    }
  }

  await Promise.all(pool); // 等待最后一批任务完成
}

function debounce（func,delay）{
    let timer = null
    
    return function(...args){
        const context = this // 保留上下文
        if(timer) clearTimeOut(timer)
        timer = setTimeout(() => {
        func.apply(context, args);
        timer = null; // 执行后清空定时器
        }, delay);
    }
}

//     childen: [{

// }]

function flattenToTree(list) {
    // 创建映射表存储每个节点，便于快速查找
    const nodeMap = {};
    // 存储最终的树形结构
    const tree = [];

    // 第一步：构建节点映射表
    list.forEach(node => {
        // 为每个节点初始化children数组
        nodeMap[node.id] = { ...node, children: [] };
    });

    // 第二步：构建树形结构
    list.forEach(node => {
        const currentNode = nodeMap[node.id];
        // 如果是根节点(pid为0)，直接加入树中
        if (node.pid === 0) {
            tree.push(currentNode);
        } else {
            // 找到父节点并将当前节点加入父节点的children
            const parentNode = nodeMap[node.pid];
            if (parentNode) {
                parentNode.children.push(currentNode);
            }
        }
    });

    return tree;
}
