  2. 根据权限筛选路由：在 Pinia 中根据用户角色 / 权限筛选asyncRoutes，生成用户可访问的路由表，再通过router.addRoute动态注册
// store/permission.js
const usePermissionStore = defineStore('permission', {
  actions: {
    generateRoutes() {
      const { role, permissions } = useAuthStore(); // 获取用户权限
      // 筛选有权访问的路由
      const accessibleRoutes = asyncRoutes.filter(route => {
        // 角色校验：路由允许的角色包含用户角色
        const hasRole = route.meta.role.includes(role);
        // 权限标识校验：用户权限包含路由所需权限
        const hasPermission = permissions.includes(route.meta.permission);
        return hasRole && hasPermission;
      });
      // 动态注册路由
      accessibleRoutes.forEach(route => router.addRoute(route));
      return accessibleRoutes;
    }
  }
});

router.beforeEach(async (to, from, next) => {
  const { token } = useAuthStore();
  const { generateRoutes } = usePermissionStore();
  if (token) {
    // 已登录：生成权限路由后检查是否有权访问
    const accessibleRoutes = await generateRoutes();
    const isAccessible = accessibleRoutes.some(route => route.path === to.path);
    if (isAccessible) {
      next();
    } else {
      next('/403'); // 无权限跳转403
    }
  } else {
    // 未登录：跳转登录页
    next('/login');
  }
});

85. Axios 的二次封装
1. 基础封装：构建请求模块的“骨架”
  1. 创建Axios实例，配置基础参数
  2. 请求拦截器：统一添加 token，处理请求前逻辑
  3. 响应拦截器：统一处理响应、错误和状态码
    1. 处理业务响应格式
    2. 处理HTTP状态码错误
// src/utils/request.js
import axios from 'axios';
import { ElMessage, ElLoading } from 'element-plus';
import { useAuthStore } from '@/store/auth';

// 1. 创建Axios实例，配置基础参数
const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL, // 从环境变量取baseURL，支持多环境
  timeout: 15000, // 超时时间15秒（覆盖Axios默认的10秒，适配算法接口的慢响应场景）
  withCredentials: true // 允许跨域请求携带Cookie（解决单点登录的身份传递问题）
});

// 2. 请求拦截器：统一添加token、处理请求前逻辑
request.interceptors.request.use(
  (config) => {
    // 2.1 自动添加token（从Pinia中获取，无需每个请求手动传）
    const authStore = useAuthStore();
    if (authStore.token) {
      config.headers.Authorization = `Bearer ${authStore.token}`;
    }

    // 2.2 大型请求显示加载中（如算法接口调用，用户感知操作状态）
    if (config.showLoading !== false) {
      config.loadingInstance = ElLoading.service({ text: '处理中...' });
    }

    return config;
  },
  (error) => {
    // 请求发出前的错误（如参数错误），直接提示
    ElMessage.error('请求参数错误，请检查');
    return Promise.reject(error);
  }
);

// 3. 响应拦截器：统一处理响应、错误和状态码
request.interceptors.response.use(
  (response) => {
    // 3.1 关闭加载中状态
    response.config.loadingInstance?.close();

    // 3.2 处理业务响应格式（后端统一返回 { code, data, message }）
    const { code, data, message } = response.data;
    if (code === 200) {
      // 成功：直接返回data，简化业务层取值（无需每次写response.data.data）
      return data;
    } else {
      // 业务错误（如参数无效、权限不足），提示错误信息
      ElMessage.error(message || '操作失败');
      return Promise.reject(new Error(message || '业务错误'));
    }
  },
  (error) => {
    // 3.3 关闭加载中状态（防止失败时loading不消失）
    error.config?.loadingInstance?.close();

    // 3.4 处理HTTP状态码错误
    const status = error.response?.status;
    switch (status) {
      case 401:
        // token过期/未登录：清除状态并跳转登录页
        const authStore = useAuthStore();
        authStore.clearToken();
        window.location.href = '/login';
        ElMessage.error('登录已过期，请重新登录');
        break;
      case 403:
        ElMessage.error('无权限执行此操作');
        break;
      case 500:
        ElMessage.error('服务端异常，请稍后重试');
        break;
      case 408:
        ElMessage.error('请求超时，请重试');
        break;
      default:
        ElMessage.error('网络错误，请检查连接');
    }
    return Promise.reject(error);
  }
);

export default request;

86. 接口联调出现问题，怎么解决
  1. 请求是否发出——看控制台/Network
✅ 排查步骤：
    - ✅ 检查请求 URL 是否拼错（如 /api 少斜杠）
    - ✅ 检查 baseURL 是否配置正确
    - ✅ 确认是否遗漏了 return / await
    - ✅ 若使用 async/await，确保函数被调用
    - ✅ 看控制台是否有 JS 报错导致代码未执行到请求位置
  2. 请求发出但失败（HTTP 层面）—— 状态码，msg
✅ 排查思路：
    - 打开 Network → Headers
      - 确认 Method、URL、参数、Header 是否与接口文档一致
    - 检查 Request Payload / Query Params
      - 参数名、类型、层级是否正确
    - 检查 Response
      - 是否返回 JSON？是否 code != 0？
  3. 数据格式错误（后端返回异常）
表现：
    - 返回空对象或字段不符
    - 前端 undefined、TypeError: Cannot read property 'xx'
    - 数据渲染错误
✅ 排查步骤：
    - 打印接口返回结果：
    - console.log(res)
    - 对照接口文档，确认字段名、层级、类型是否变化
    - 如果接口更新，和后端同步更新前端解析逻辑
  4. 数据正常但页面没显示（渲染逻辑错误）
❗表现：
    Network 返回 200 且数据正确，但页面空白
✅ 排查思路：
    - 检查是否异步未完成前直接渲染（如未等待 await）
    - Vue 中看 ref/reactive 是否响应式
    - React 中看 setState 是否在正确位置调用
    - 模板中字段名是否拼错（例如 userInfo.name 写成 user.name）
  5. 排查顺序
1️⃣ 看控制台是否报错（JS 层面）
2️⃣ 看 Network 是否发出请求
3️⃣ 看请求地址与参数是否正确
4️⃣ 看响应状态码与返回内容
5️⃣ 看页面是否正确处理数据
6️⃣ 若都正常仍异常 → 用 Postman 测接口、清缓存、查代理

  1. 用于解决浏览器的默认样式差异，它通过清除浏览器的默认样式来保证不同浏览器的样式一致性。

  3. ress的好处：
    1. 提高开发效率，增强可维护性
    2. 改善用户体验，支持现代Web标准
  基本原理
ress 会重置一些常见的 HTML 元素的样式，使它们在所有浏览器中看起来一致：
  - 移除元素的边距和填充（margin 和 padding）
  - 设置字体为通用字体
  - 将所有图片和视频的 max-width 设置为 100%，以防止它们超出容器
  - 为所有元素设置 box-sizing: border-box;，让它们的宽度和高度更易于控制

97. Monorepo（多包仓库）管理
1. 定义：Monorepo（Monolithic Repository）→ 是一种「单一代码仓库管理多个项目或包」的代码管理模式。
2. 使用pnpm workspace实现
  1. 根目录下package.json 中，private: true 是必须的，否则 pnpm 不允许 workspace 管理多个包
  2. 配置 workspace pnpm-workspace.yaml
// 告诉pnpm 所有位于 packages/ 目录下的子项目，都是一个 workspace 包。
packages:
  - "packages/*"
  3. 子包的 package.json 会有："dependencies": { "@my/utils": "workspace:*" }，表示，使用本地的包，不是从 npm 拉取
  4. 安装依赖：pnpm i
    - 自动解析 workspace 依赖关系
    - 在全局虚拟 store 中安装依赖（节省空间）
    - 在子包之间建立符号链接（symlink）

99. 什么是UA容器
1. 定义：UA（User Agent）是浏览器或 App 在每次发送 HTTP 请求时带上的一段标识字符串。
2. 实际使用：UA容器就是网页运行的环境（比如微信、支付宝、App 内、Safari）。我们判断它，是为了做「环境兼容」和「功能适配」。
